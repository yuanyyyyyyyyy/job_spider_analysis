2 mysql8 数据库名为xm
create table jobs (
  jid int(10)  auto_increment,
  job_name varchar(200), -- 工作名称
  company_name varchar(200), -- 公司名称
  job_sal varchar(200), -- 工资
  job_loc varchar(200), -- 工作地址
  job_exp varchar(400), -- 工作经验要求
  job_edu varchar(200) ,-- 学历要求
  primary key(id)
)
